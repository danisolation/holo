```typescript
export interface SectorDetailTickerItem {
  symbol: string;
  name: string;
  industry: string | null;
  close: number | null;
  volume: number | null;
  change_7d: number | null;
  change_30d: number | null;
  market_cap: number | null;
}

export interface SectorDetailResponse {
  sector: string;
  ticker_count: number;
  tickers: SectorDetailTickerItem[];
}

export interface PeerComparisonItem {
  symbol: string;
  name: string;
  close: number | null;
  volume: number | null;
  change_1d: number | null;
  pe: number | null;
  market_cap: number | null;
  rank_pe: number | null;
  rank_volume: number | null;
  rank_change: number | null;
  rank_market_cap: number | null;
  is_target: boolean;
}

export interface PeerComparisonResponse {
  symbol: string;
  sector: string;
  peers: PeerComparisonItem[];
}

// Hooks:
export function useSectorDetail(sectorName: string | undefined): UseQueryResult<SectorDetailResponse>;
export function usePeerComparison(symbol: string | undefined): UseQueryResult<PeerComparisonResponse>;
```
