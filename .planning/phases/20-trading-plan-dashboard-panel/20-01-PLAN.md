3. Add `fetchTradingSignal` function AFTER the existing `fetchAnalysisSummary` function:

```typescript
export async function fetchTradingSignal(
  symbol: string,
): Promise<TickerTradingSignal | null> {
  try {
    const result = await apiFetch<
      AnalysisResult & { raw_response?: TickerTradingSignal }
    >(`/analysis/${encodeURIComponent(symbol)}/trading-signal`);
    return result.raw_response ?? null;
  } catch (e) {
    if (e instanceof ApiError && e.status === 404) return null;
    throw e;
  }
}
```

This follows the exact same pattern as `fetchAnalysisSummary`. The endpoint returns `AnalysisResult` shape but with `raw_response` containing the full `TickerTradingSignal`. The function extracts `raw_response` and returns it, or `null` if the endpoint 404s (no trading signal computed yet).
