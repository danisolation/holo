**Step 2: Create `frontend/playwright.config.ts`**

```typescript
import { defineConfig, devices } from '@playwright/test';
import path from 'path';

export default defineConfig({
  testDir: './e2e',
  fullyParallel: true,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 2 : 0,
  workers: process.env.CI ? 1 : undefined,
  reporter: 'html',

  use: {
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
    screenshot: 'only-on-failure',
    video: 'retain-on-failure',
  },

  expect: {
    toHaveScreenshot: {
      maxDiffPixelRatio: 0.01,
      threshold: 0.2,
      animations: 'disabled',
    },
  },

  webServer: [
    {
      command: '.venv/Scripts/python -m uvicorn app.main:app --port 8001',
      cwd: path.resolve(__dirname, '..', 'backend'),
      url: 'http://localhost:8001/',
      reuseExistingServer: !process.env.CI,
      timeout: 60_000,
      env: {
        HOLO_TEST_MODE: 'true',
      },
    },
    {
      command: 'npx next dev --port 3000',
      url: 'http://localhost:3000',
      reuseExistingServer: !process.env.CI,
      timeout: 60_000,
      env: {
        NEXT_PUBLIC_API_URL: 'http://localhost:8001/api',
      },
    },
  ],

  projects: [
    {
      name: 'chromium',
      use: { ...devices['Desktop Chrome'] },
    },
  ],
});
```

Key config decisions:
- Backend webServer uses `cwd` pointing to `../backend` so pydantic-settings finds `.env` file
- Backend command uses Windows venv path: `.venv/Scripts/python` (this is a Windows dev environment)
- `HOLO_TEST_MODE: 'true'` is passed via webServer env to backend process
- `NEXT_PUBLIC_API_URL: 'http://localhost:8001/api'` ensures frontend calls test backend (port 8001, not 8000)
- Chromium only (no WebKit on Windows per research findings)
- `reuseExistingServer: !process.env.CI` — reuse in dev for speed, fresh in CI for clean state
- 60s timeout for server startup (generous for cold start)

```typescript
import { test, expect } from '@playwright/test';

test.describe('Infrastructure smoke tests', () => {
  test('backend health check returns ok', async ({ request }) => {
    const response = await request.get('http://localhost:8001/');
    expect(response.ok()).toBeTruthy();
    const body = await response.json();
    expect(body.status).toBe('ok');
    expect(body.service).toBe('holo');
  });

  test('frontend homepage loads', async ({ page }) => {
    await page.goto('/');
    await expect(page.locator('body')).toBeVisible();
    // Verify the app rendered (not a blank page or error)
    await expect(page.locator('text=Holo').first()).toBeVisible();
  });
});
```
