```python
"""Pydantic schemas for health monitoring API endpoints."""
from pydantic import BaseModel


class JobStatusItem(BaseModel):
    """Single job's latest execution status."""
    job_id: str
    job_name: str
    status: str          # success, partial, failed
    color: str           # green, yellow, red
    started_at: str | None   # ISO datetime
    completed_at: str | None # ISO datetime
    duration_seconds: float | None
    result_summary: dict | None
    error_message: str | None


class JobStatusResponse(BaseModel):
    """GET /health/jobs response."""
    jobs: list[JobStatusItem]


class DataFreshnessItem(BaseModel):
    """Single data source freshness."""
    data_type: str       # Vietnamese label e.g. "Giá cổ phiếu"
    table_name: str      # e.g. "daily_prices"
    latest: str | None   # ISO datetime of latest record
    is_stale: bool
    threshold_hours: int


class DataFreshnessResponse(BaseModel):
    """GET /health/data-freshness response."""
    items: list[DataFreshnessItem]


class ErrorRateDayItem(BaseModel):
    """Error count for a single day."""
    day: str             # ISO date
    total: int
    failed: int


class ErrorRateJobItem(BaseModel):
    """Error rate for a single job over 7 days."""
    job_id: str
    job_name: str
    days: list[ErrorRateDayItem]
    total_runs: int
    total_failures: int


class ErrorRateResponse(BaseModel):
    """GET /health/errors response."""
    jobs: list[ErrorRateJobItem]


class DbPoolResponse(BaseModel):
    """GET /health/db-pool response."""
    pool_size: int
    checked_in: int
    checked_out: int
    overflow: int
    max_overflow: int


class TriggerResponse(BaseModel):
    """POST /health/trigger/{job_name} response."""
    message: str
    triggered: bool
```

Implement each method:

**`get_job_statuses()`** — per D-10-03 (green/yellow/red from job_executions):
- Use PostgreSQL `DISTINCT ON (job_id) ... ORDER BY job_id, started_at DESC` to get the latest execution per job_id (per RESEARCH.md Pitfall 2).
- For each row, compute:
  - `color`: `{"success": "green", "partial": "yellow", "failed": "red"}.get(status, "red")`
  - `job_name`: Look up from `_JOB_NAMES` dict (imported from manager.py). Use `job_id.replace("_", " ").title()` as fallback.
  - `duration_seconds`: `(completed_at - started_at).total_seconds()` if both exist, else None
  - Format `started_at` and `completed_at` as `.isoformat()` strings
  - Pass through `result_summary` and `error_message` as-is
- Return list of dicts matching `JobStatusItem` fields.

**`get_data_freshness()`** — per D-10-02:
- Define a list of tuples: `(table_name, label_vi, timestamp_column, threshold_hours)`:
  - `("daily_prices", "Giá cổ phiếu", "date", 48)`
  - `("technical_indicators", "Chỉ báo kỹ thuật", "date", 48)`
  - `("ai_analyses", "Phân tích AI", "analysis_date", 48)`
  - `("news_articles", "Tin tức", "created_at", 48)`
  - `("financials", "Báo cáo tài chính", "created_at", 168)`
- For each table, execute `text(f"SELECT MAX({col}) as latest FROM {table}")`. NOTE: table names and column names are hardcoded constants — NEVER from user input (per RESEARCH.md security pattern).
- Compute staleness: If `latest` is a `date` (no `.hour` attribute), convert to age in hours via `(date_type.today() - latest).days * 24`. If it's a `datetime`, use `(datetime.now(timezone.utc) - latest).total_seconds() / 3600`. Mark `is_stale = True` if age > threshold_hours or if latest is None.
- Format `latest` as `.isoformat()` string (or None).
- Return list of dicts matching `DataFreshnessItem` fields.

**`get_error_rates(days=7)`** — per D-10-03 (HEALTH-03):
- Query `job_executions` for all rows where `started_at >= now - {days} days`.
- Use: `SELECT job_id, DATE(started_at AT TIME ZONE 'UTC') as day, COUNT(*) as total, COUNT(*) FILTER (WHERE status = 'failed') as failed FROM job_executions WHERE started_at >= :since GROUP BY job_id, day ORDER BY job_id, day`
- Group results by `job_id` into a dict. For each job_id, build the list of `ErrorRateDayItem` dicts and compute `total_runs` and `total_failures` (sums).
- Look up `job_name` from `_JOB_NAMES` (same as get_job_statuses).
- Return list of dicts matching `ErrorRateJobItem` fields.

```python
@router.post("/trigger/{job_name}", response_model=TriggerResponse)
async def trigger_job(job_name: str):
    """Manually trigger a pipeline job. Per D-10-04."""
    if job_name not in JOB_TRIGGER_MAP:
        raise HTTPException(
            status_code=404,
            detail=f"Unknown job: '{job_name}'. Valid: {', '.join(sorted(JOB_TRIGGER_MAP.keys()))}"
        )
    manual_id, func_name = JOB_TRIGGER_MAP[job_name]

    # Lazy import to avoid circular deps (same pattern as manager.py)
    from app.scheduler import jobs as job_module
    job_func = getattr(job_module, func_name)

    scheduler.add_job(
        job_func,
        id=manual_id,
        replace_existing=True,
        misfire_grace_time=3600,
    )
    logger.info(f"Manual trigger: {job_name} → {manual_id}")
    return TriggerResponse(message=f"Job '{job_name}' triggered", triggered=True)
```

**Endpoint 6: GET /health/summary** — overall health summary (per RESEARCH.md Open Question 1 — new endpoint, keep existing /health as-is):
```python
@router.get("/summary")
async def get_health_summary():
    """Overall health summary aggregating jobs + freshness + pool."""
    async with async_session() as session:
        svc = HealthService(session)
        jobs = await svc.get_job_statuses()
        freshness = await svc.get_data_freshness()

    pool = engine.pool
    job_colors = [j["color"] for j in jobs]
    stale_count = sum(1 for f in freshness if f["is_stale"])

    # Overall status: red if any job red or any data stale, yellow if any yellow, else green
    if "red" in job_colors or stale_count > 0:
        overall = "degraded"
    elif "yellow" in job_colors:
        overall = "warning"
    else:
        overall = "healthy"

    return {
        "status": overall,
        "jobs_total": len(jobs),
        "jobs_healthy": job_colors.count("green"),
        "jobs_warning": job_colors.count("yellow"),
        "jobs_error": job_colors.count("red"),
        "data_sources_total": len(freshness),
        "data_sources_stale": stale_count,
        "pool_checked_out": pool.checkedout(),
        "pool_available": pool.size() - pool.checkedout() + (3 - pool.overflow()),
    }
```
