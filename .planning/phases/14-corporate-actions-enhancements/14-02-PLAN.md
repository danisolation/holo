   class ExDateAlertService:
       def __init__(self, session: AsyncSession):
           self.session = session

       async def check_upcoming_exdates(self, chat_id: str | None = None) -> int:
           """Find events with ex_date in next 3 business days for held+watched tickers. Send alerts."""
   ```
   Implementation details:
   - Get target_chat from chat_id or settings.telegram_chat_id. Return 0 if no chat_id.
   - Compute date range: today → today + 5 calendar days (covers 3 business days accounting for weekends). Then filter to only Mon-Fri ex_dates in that range.
   - Query corporate_events WHERE:
     - `alert_sent = False`
     - `ex_date BETWEEN today AND today+5 days`
     - `ex_date` weekday is Mon-Fri (0-4)
   - Get watchlisted ticker_ids: `SELECT ticker_id FROM user_watchlist WHERE chat_id = target_chat`
   - Get held ticker_ids: Use a subquery on trades grouped by ticker_id where net quantity > 0 (SUM of BUY quantity - SUM of SELL quantity > 0). Use Lot model if available, otherwise use Trade model with CASE WHEN side='BUY' THEN quantity ELSE -quantity END grouped by ticker_id HAVING sum > 0.
   - Combine: `ticker_ids = watchlisted_ids | held_ids`
   - Filter events to only those with ticker_id in combined set
   - For each matching event: format message via MessageFormatter.exdate_alert, send via telegram_bot.send_message, mark alert_sent=True, commit
   - Return count of alerts sent
   - Never raise — follow the pattern in AlertService (catch all, log, return partial count)

   async def daily_exdate_alert_check():
       """Check for upcoming ex-dates and send Telegram alerts.

       Triggered after daily_corporate_action_check via job chaining.
       Never raises — alert failure must not break the pipeline.
       """
       logger.info("=== DAILY EXDATE ALERT CHECK START ===")
       async with async_session() as session:
           job_svc = JobExecutionService(session)
           execution = await job_svc.start("daily_exdate_alert_check")
           try:
               from app.services.exdate_alert_service import ExDateAlertService
               service = ExDateAlertService(session)
               result = await service.check_upcoming_exdates()
               await job_svc.complete(
                   execution, status="success",
                   result_summary={"alerts_sent": result},
               )
               await session.commit()
               logger.info(f"=== DAILY EXDATE ALERT CHECK COMPLETE: {result} alerts sent ===")
           except Exception as e:
               await job_svc.complete(
                   execution, status="partial",
                   result_summary={"error": str(e)[:200]},
               )
               await session.commit()
               logger.error(f"=== DAILY EXDATE ALERT CHECK FAILED: {e} ===")
   ```
   Pattern matches daily_signal_alert_check exactly — never raises.
